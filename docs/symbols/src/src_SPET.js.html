  9 /**
 10  * Syncapse Platform Event Track 1.1
 11  * 
 12  * Event track is a system for tracking events in Facebook Tabs and Applications.  
 13  * These actions are tracked using a client-side library and recorded within the Syncapse Platform. 
 14  * Clients with access to the Measure product within the Syncapse Platform can run reports on the tracked events.
 15  * 
 16  * 
 17  * @class Represents the Syncapse Platform Event Tracker
 18  * @param {String} source The URL of the App/Tab that is being tracked
 19  * @param {String} [session] The current session
 20  * @param {Object} [params] any parameters to be passed to the "page-view" event (also can include debug: true)
 21  * @param {Boolean} [params.debug="true"] If debug is true, no data will be sent to the Syncapse tracking server. Instead, all tracking event data will be output to the browser console using console.log.
 22  * @param {String} [params.trackDomain="track.platform.syncapse.com"] The tracking domain to which the Event Track class will send event data.
 23  * @param {String} [params.trackPath="/track.gif"] The path component for the tracking image.
 24  * @param {String} [params.location="document.location.href"] A 'location' data element to send to the server.
 25  * @param {String} [params.fromVariableName="from"] Event Track can track where a user has arrived from by using a named GET variable.
 26  * @param {Boolean} [params.inFbml="false"] If you are using Event Track in an FBML context (which is officially unsupported), you may want to set the parameter to 'true'.
 27  * @param {String} [params.imageId="spet_image"] If you are using Event Track in an FBML context (officially unsupported), Event Track needs a DOM image element with this ID.
 28  * @param {Boolean} [params.skipOnLoadEvent="false"] Event Track will fire a 'page-view' event automatically in the constructor. If this is not your desired behaviour, set this parameter to 'true'.
 29  * 
 30  * @returns {SPET} An instance of the Syncapse Platform Event Tracker.
 31  */
 32 function SPET(source, session, params) 
 33 {
 34 	// Set the debug flag
 35 	this.debug = (typeof params.debug !== 'undefined' && params.debug !== false);
 36 	delete params.debug;
 37 	
 38 	// An explicit 'source' variable is required
 39 	if (typeof source === 'undefined' || ! source) {
 40 		throw new Error("You must pass a source URL string to the SPET constructor");
 41 	}
 42 	
 43 	// Add a session if one doesn't exist - a 'session' is a generic way of grouping events for further use
 44 	session = session || this.generateSession();
 45 	
 46 	// Store source and session for later
 47 	this.source = source;
 48 	this.session = session;
 49 	
 50 	// If no params are set, use some defaults
 51 	params = params || this.getDefaultParams();
 52 	
 53 	// Set the client name
 54 	params.client = 'SPET JavaScript/1.1';
 55 	
 56 	// Add a default tracking domain
 57 	// If params have a trackDomain, use that instead of our default
 58 	this.trackDomain = params.trackDomain || 'track.platform.syncapse.com';
 59 	
 60 	// Set the default location
 61 	params.location = params.location || document.location.href;
 62 	
 63 	// Get query string parameters
 64 	params = this.addQueryVars(params, document.location.search ? document.location.search.substring(1) : "");
 65 	
 66 	// Populate a 'from' var - the query string variable name can be specified in the params as 'fromVariableName'
 67 	var fromVariableName = params.fromVariableName || 'from';
 68 	params.from = params["get_" + fromVariableName];
 69 	
 70 	// See if we are in FBML
 71 	this.inFbml = ((typeof params.inFbml !== 'undefined') && params.inFbml) || (typeof Image === 'undefined');
 72 	
 73 	// If in FBML, we probably want to know the image bug element's DOM ID
 74 	this.imageId = params.imageId || 'spet_image';
 75 	
 76 	// Allow overriding the trackPath along with the trackDomain
 77 	this.trackPath = params.trackPath = params.trackPath || '/track.gif'; 
 78 	
 79 	// Store params for re-use later.
 80 	this.baseParams = params;
 81 	
 82 	// Skip the automatic page-view event?
 83 	params.skipOnLoadEvent = (typeof params.skipOnLoadEvent !== 'undefined') && params.skipOnLoadEvent;
 84 	
 85 	// If we're not in FBML, we can trigger JavaScript without any user interaction
 86 	if (!this.inFbml && !params.skipOnLoadEvent) {
 87 		this.track('page-view', params);
 88 	};
 89 };

 91 /**
 92  * Add query vars to params object with a prefix.
 93  * 
 94  * @param {Object} parameter Parameter object
 95  * @param {String} query Query string
 96  * @param {String} [prefix="get_"] Prefix to prepend to query var names
 97  * @returns {Object} Modified parameter object
 98  */
 99 SPET.prototype.addQueryVars = function(params, query, prefix) {
100 	var q = this.getQueryParams(query);
101 	prefix = prefix || "get_";
102 	for (var k in q) {
103 		params[prefix + k] = q[k];
104 	}
105 	return params;
106 };

108 /**
109  * Get the default parameter object
110  * @returns {Object} Default parameter object
111  */
112 SPET.prototype.getDefaultParams = function() {
113 	return {
114 		// Set a default location
115 		location: document.location.href,
116 		
117 		// If debug is true, we'll output event to the browser console
118 		debug: true,
119 		
120 		// Sometimes we may want to skip the on load 'page-view' event
121 		skipOnLoadEvent: false,
122 		
123 		// Allow explicit control over if we are in FBML
124 		inFbml: false
125 	};
126 };

136 /**
137  * Generates a new session (based on the time and a random number)
138  * Note: has no side-effect (aka the session is not stored in the object)
139  * @returns {String} the generated session string
140  */
141 SPET.prototype.generateSession = function() {
142 	var time = (new Date()).getTime();
143 	var random = Math.floor(Math.random() * 10000000);	
144 	return time + '-' + random;
145 };

147 /**
148  * Get the URL used to track events. Uses the 'trackDomain' and 'trackPath' parameters, 
149  * and is fully HTTPS-aware.
150  * 
151  * @returns {String} The base tracking image URL
152  */
153 SPET.prototype.getUrl = function() {
154 	var protocol = 'http';
155 	try {
156 		if (document.location.protocol === 'https:') {
157 			protocol = 'https';
158 		}
159 	} catch (ex) { /* noop */ }
160 	
161 	return protocol + "://" + this.trackDomain + this.trackPath;
162 };

164 /**
165  * Get query string vars in an object. This is non-smart, and doesn't handle
166  * '[]' notation or do anything clever.
167  * 
168  * @param {String} query
169  * @returns {Object} A hash of key => value pairs from the query string vars
170  */
171 SPET.prototype.getQueryParams = function(query) {
172     var qs = {},
173     params = query.split("&"), parts;
174     for(var i = 0; i < params.length; i++){
175         parts = params[i].split("=");
176         qs[ parts[0] ] = parts[1] ? parts[1] : "";
177     } 
178     return qs;
179 };

181 /**
182  * Builds a query string from a non-nested object
183  * @param {Object} obj the object the querystring will be built from
184  * @returns {String} the querystring
185  */
186 SPET.prototype.buildQueryString = function(obj) {
187 	if (this.inFbml) {
188 		return this.buildFbmlQueryString(obj);
189 	}
190 	var rv = '?';
191 	var key;
192 	for(key in obj) {
193 		if (typeof(obj[key]) !== 'function') {
194 			if (rv.length > 1) {
195 				rv += '&';
196 			}
197 			rv += encodeURIComponent(key) + '=' + encodeURIComponent("" + obj[key]);
198 		}
199 	}
200 	return rv.replace(/%20/g, '+');
201 };
202 
203 /**
204  * Builds a query string from a non-nested object. Legacy FBML/FBJS-compatible.
205  * @param {Object} obj the object the querystring will be built from
206  * @returns {String} the querystring
207  */
208 SPET.prototype.buildFbmlQueryString = function(obj) {
209 	var pairs = [];
210 	for (var k in obj) {
211 		if (typeof(obj[k]) == 'function') {
212 			continue;
213 		}
214 		// Stringify
215 		var value = obj[k] + "";
216 		// Deal with common values that should really be null or blank
217 		if (value == "null") value = "";
218 		if (value == "undefined") value = "";
219 		// Add our key value pair
220 		pairs.push(escape(k) + '=' + escape(obj[k]));
221 	}
222 	return '?' + pairs.join('&');
223 };

237 /**
238  * Track an event by loading a dummy image source URL.
239  * If the debug parameter is true, this method will output event data to the browser console.
240  * 
241  * @param {String} event the event to track
242  * @param {Object} [params] the parameter object
243  */
244 SPET.prototype.track = function(event, params) {
245 	try {
246 		params = params || {};
247 		params.se = this.session;
248 		params.so = this.source;
249 		params.ev = event;
250 		params.r = Math.floor(Math.random() * 10000000);
251 		
252 		// Merge the event-specific params with our base params.
253 		params = this.extend({}, this.baseParams, params);
254 		
255 		// We don't need to track some params
256 		var untracked = [ 'trackPath', 'trackDomain', 'skipOnLoadEvent' ];
257 		for (var k in untracked) {
258 			delete params[untracked[k]];
259 		}
260 		
261 		// Build a full image source URL for tracking 
262 		var trackUrl = this.getUrl() + this.buildQueryString(params);
263 		
264 		if (!this.debug) {
265 			// Set the image source
266 			this.setImageSource(trackUrl);
267 		} else {
268 			// Add the full trackUrl to help debugging
269 			params.trackUrl = trackUrl;
270 			this.log(event, params);
271 		}
272 	} catch (ex) {
273 		this.log("error-lib", {error: "" + ex});
274 	}
275 	
276 	return true;
277 };

279 /**
280  * Set an image's src attribute. Legacy FBML/FBJS-compatible.
281  * @param {String} url Full image source URL
282  */
283 SPET.prototype.setImageSource = function(url) {
284 
285 	// Deal with the non-legacy-FBML/FBJS case first and get outta here
286 	if (!this.inFbml) {
287 		try {
288 			var image = new Image();
289 			image.src = url;
290 		} catch (ex) {
291 			this.log("error-lib", {error: "" + ex});
292 		}
293 		return;
294 	}
295 
296 	// We're in Legacy FBML/FBJS and can't call 'new Image()'. Instead, look for an existing image in the DOM.
297 	var img = document.getElementById(this.imageId);
298 	if (!img) {
299 		var ex = "Cannot find Image element with id '" + this.imageId + "'. You must embed an Image element into your FBML page.";
300 		this.log("error-lib", {error: "" + ex});
301 		return;
302 	}
303 	
304 	try {
305 		// Set the Image source using FBJS
306 		img.setSrc(url);
307 	} catch (ex) {
308 		this.log("error-lib", {error: "" + ex});
309 	}
310 };

312 /**
313  * Automatically attach listeners to children of an element
314  * @param {DOMElement} elem the containing element we're binding events within (typically BODY)
315  */
316 SPET.prototype.attachListeners = function(elem) {
317 	if (typeof($) == 'undefined') {
318 		return; // Sorry, this won't work without a nice $ (jQuery)
319 	};
320 	var spet = this;
321 	$(elem).find('a[track!="false"]').mousedown(function() { spet.trackElement(this); });
322 	$(elem).find('form[track!="false"]').submit(function() { spet.trackElement(this); });
323 };

325 /**
326  * Automatically attach listeners to the Facebook API.
327  * Requires that the Facebook JavaScript SDK is loaded
328  * @see https://developers.facebook.com/docs/reference/javascript/FB.Event.subscribe/ 
329  */
330 SPET.prototype.attachFBListeners = function() {
331 	if (typeof(FB) == 'undefined') return; // Sorry, this won't work without the FB namespace
332 	var spet = this;
333 	try {
334 		
335 		// Track Comment creation
336 		FB.Event.subscribe("comment.create", function(response) { 
337 			spet.track("comment-create"); 
338 		});
339 		// Track Comment removal
340 		FB.Event.subscribe("comment.remove", function(response) { 
341 			spet.track("comment-remove"); 
342 		});		
343 		
344 		// Track Likes
345 		FB.Event.subscribe("edge.create", function(href) { 
346 			spet.track("like-create", {
347 				"target": href
348 			}); 
349 		});
350 		// Track "Unlikes"		
351 		FB.Event.subscribe("edge.remove", function(href) { 
352 			spet.track("like-remove", {
353 				"target": href
354 			});
355 		});		
356 		
357 		// Track login/logout
358 		FB.Event.subscribe("auth.authResponseChange", function(response) {
359 			if (response.status == 'connected') {
360 				spet.track("facebook-login", {
361 					"facebook_uid": response.authResponse.userID
362 				}); 
363 			} else if (response.status == 'unknown') {
364 				spet.track("facebook-logout");
365 			}
366 		});
367 	} catch (ex) {
368 		spet.track("error-facebook", {error: ex.toString()});
369 	}
370 };

372 /**

374  * @see attachFBListeners
375  */
376 SPET.prototype.attachFbListeners = function() { 
377 	return this.attachFBListeners();

381 /**
382  * Handle tracking an A tag click, or a FORM tag submit
383  * @param {DOMElement} element the element we are tracking
384  */
385 SPET.prototype.trackElement = function(element) {
386 	if (typeof($) == 'undefined') return; // Sorry, this won't work without a nice $
387 	var tagName = element.tagName.toLowerCase();
388 	var event = (tagName === 'form') ? 'form-submit' : 'link-click';
389 	
390 	var jElement = $(element);
391 	if (jElement.attr('track')) {
392 		event = jElement.attr('track');
393 	}
394 	var params = { 'element': tagName };
395 	if (jElement.attr('href')) {
396 		params.url = jElement.attr('href');
397 	}
398 	if (tagName === 'form') {
399 		if (jElement.attr('action')) {
400 			params.url = jElement.attr('action');
401 		}
402 		params.data = $(element).serialize();
403 	}
404 	return this.track(event, params);
405 };

407 /**
408  * Extend an object. Legacy FBML/FBJS compatible.
409  * @param {Object} obj Base object
410  * @param {Object} extObj Object with properties to extend onto obj
411  * @returns {Object} obj, extended by extObj
412  */
413 SPET.prototype.extend = function(obj, extObj) {
414     if (arguments.length > 2) {
415         for (var a = 1; a < arguments.length; a++) {
416             this.extend(obj, arguments[a]);
417         }
418     } else {
419         for (var i in extObj) {
420             obj[i] = extObj[i];
421         }
422     }
423     return obj;
424 };
